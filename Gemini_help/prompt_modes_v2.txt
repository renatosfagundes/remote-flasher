// CombustionManualMode.qml
import QtQuick 2.15
import QtQuick.Layouts 1.15

Item {
    id: root
    property real s: 1.0

    width: 1200 * s
    height: 650 * s

    Rectangle {
        anchors.fill: parent
        color: "#101616"
        radius: 10 * s
        border.color: "#252540"
        border.width: 1 * s
    }

    RowLayout {
        anchors.fill: parent
        anchors.margins: 40 * s
        spacing: 20 * s

        // RPM Gauge on the LEFT
        Gauge {
            id: rpmGauge
            s: root.s
            Layout.preferredWidth: parent.width * 0.35
            Layout.fillHeight: true
            value: dashboard ? dashboard.rpm : 0
            maximumValue: 8000; labelStepSize: 1000; minorTickStep: 500
            unitLabel: "x1000"; accentColor: "#3498db"; redZoneStart: 6500
            Behavior on value { NumberAnimation { duration: 400 } }
        }

        // Centre: H-PATTERN SHIFT DIAGRAM
        Canvas {
            Layout.preferredWidth: parent.width * 0.2
            Layout.fillHeight: true
            onPaint: {
                var ctx = getContext("2d");
                ctx.reset();
                ctx.save();

                var cx = width / 2;
                var cy = height / 2;

                ctx.lineWidth = 3 * root.s;
                ctx.strokeStyle = "rgba(255,255,255,0.4)";
                ctx.lineCap = "round";

                var gateW = 30 * root.s;
                var gateH = 60 * root.s;
                var railY = cy;

                var x_12 = cx - gateW;
                var x_34 = cx;
                var x_56R = cx + gateW;

                // Horizontal rail
                ctx.beginPath();
                ctx.moveTo(x_12, railY);
                ctx.lineTo(x_56R, railY);
                ctx.stroke();

                // Vertical gates
                ctx.beginPath(); ctx.moveTo(x_12, railY - gateH / 2); ctx.lineTo(x_12, railY + gateH / 2); ctx.stroke();
                ctx.beginPath(); ctx.moveTo(x_34, railY - gateH / 2); ctx.lineTo(x_34, railY + gateH / 2); ctx.stroke();
                ctx.beginPath(); ctx.moveTo(x_56R, railY - gateH / 2); ctx.lineTo(x_56R, railY + gateH / 2); ctx.stroke();
                ctx.beginPath(); ctx.moveTo(x_56R + 10 * root.s, railY - gateH / 2); ctx.lineTo(x_56R + 10 * root.s, railY + gateH / 2); ctx.stroke();

                ctx.font = (16 * root.s) + "px sans-serif";
                ctx.fillStyle = "#aaa";
                ctx.textAlign = "center";
                ctx.textBaseline = "middle";

                ctx.fillText("2", x_12, railY + gateH / 2 + 10 * root.s);
                ctx.fillText("3", x_34, railY - gateH / 2 - 10 * root.s);
                ctx.fillText("4", x_34, railY + gateH / 2 + 10 * root.s);
                ctx.fillText("5", x_56R, railY - gateH / 2 - 10 * root.s);
                ctx.fillText("6", x_56R, railY + gateH / 2 + 10 * root.s);
                ctx.fillText("R", x_56R + 10 * root.s, railY - gateH / 2 - 10 * root.s);

                var currentGear = dashboard ? dashboard.gear : 0;
                if (currentGear !== 0 && currentGear !== 7 && currentGear !== 8) {
                    var px = cx;
                    var py = cy;
                    switch (currentGear) {
                        case -1: px = x_56R + 10 * root.s; py = railY - gateH / 2; break;
                        case 1: px = x_12; py = railY - gateH / 2; break;
                        case 2: px = x_12; py = railY + gateH / 2; break;
                        case 3: px = x_34; py = railY - gateH / 2; break;
                        case 4: px = x_34; py = railY + gateH / 2; break;
                        case 5: px = x_56R; py = railY - gateH / 2; break;
                        case 6: px = x_56R; py = railY + gateH / 2; break;
                    }

                    ctx.shadowBlur = 15 * root.s;
                    ctx.shadowColor = "#01E6DE";
                    ctx.fillStyle = "#01E6DE";

                    ctx.beginPath();
                    ctx.arc(px, py, 6 * root.s, 0, 2 * Math.PI);
                    ctx.fill();
                }

                ctx.restore();
            }
        }

        // Speed Gauge on the RIGHT
        Gauge {
            id: speedGauge
            s: root.s
            Layout.preferredWidth: parent.width * 0.35
            Layout.fillHeight: true
            value: dashboard ? dashboard.speed : 0
            maximumValue: 220; labelStepSize: 20; minorTickStep: 10
            unitLabel: "km/h"; accentColor: "#01E6DE"; redZoneStart: -1
            Behavior on value { NumberAnimation { duration: 600 } }
        }
    }
}

=============================================================
QML RULES TO RESPECT (common mistakes to avoid)
=============================================================
1. Repeater `model:` arrays MUST start with `[` — e.g.
   `model: [ { t: "P", v: 7 }, ... ]`
2. NEVER declare `readonly property` inside a JS function body
   (onPaint, onClicked, etc.). Inside JS use `var currentGear = ...`.
3. Test-harness Windows need `id: win` and
   `NumberAnimation { target: win; property: "X"; ... }` — never
   `target: parent` at a Window root.
4. Use `font.family: "sans-serif"` (or "Consolas") — not "Inter".
5. Always null-safe: `dashboard ? dashboard.X : defaultValue`.
